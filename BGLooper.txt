public float speed; //Toc do truot hinh anh
private Vector2 _offset = Vector2.zero;
private Material _mat;
void Start () {
	_mat = GetComponent<Renderer> ().material; //Lay thuoc tinh material
	_offset = _mat.GetTextureOffset ("_MainTex"); //Lay do doi cua hinh anh chinh
}
	
void Update () {
	_offset.y += speed * Time.deltaTime; //Tao toc do cho hinh anh
	_mat.SetTextureOffset ("_MainTex",_offset); //Cai dat toc do cho hinh anh o thuoc tinh offset
}
